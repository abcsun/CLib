352  void messenger_connection::processBlock(Array_buffer& buf, ssize_t& dataSize,
353  Msg_stream* sock)
354  {
355  uint8_t* dataPointer = buf.data();
356  ssize_t cpSize;
357 
358  if (dataSize > MESSENGER_BUFFER_SIZE)
359  VLOG_WARN(lg, "Read buffer insufficient, check MESSENGER_BUFFER_SIZE in messenger.hh");
360 
361 VLOG_DBG(lg, "***buf.data:%s,size:%zu",dataPointer,dataSize);
362  //Copy message into buffer.
363  while (dataSize > 0)
364  {
365  cpSize=msger->processBlock(dataPointer,dataSize,
366  &internalrecvbuf[0],currSize, sock);
367  if ((currSize+cpSize) > MESSENGER_MAX_MSG_SIZE)
368  VLOG_WARN(lg, "Message buffer insufficient, check MESSENGER_MAX_MSG_SIZE in messenger.hh");
369  else
370  VLOG_DBG(lg, "Copy %zu bytes to message",cpSize);
371 
372  memcpy(endpointer,dataPointer,cpSize);
373  endpointer+=cpSize;
374  dataPointer+=cpSize;
375  dataSize-=cpSize;
376  currSize+=cpSize;
377 
378  //End of message
379  if ((currSize > 0) &&
380  msger->msg_complete(&internalrecvbuf[0],currSize, sock))
381  {
382  if (MESSENGER_BYTE_DUMP)
383  {
384  fprintf(stderr,"messenger_core message of size %zu\n\t",
385  currSize);
386  uint8_t* readhead = internalrecvbuf;
387  for (int i = 0; i < currSize; i++)
388  {
389  fprintf(stderr, "%"PRIx8" ", *readhead);
390  readhead++;
391  }
392  fprintf(stderr,"\n");
393  }
394 
395  process(new core_message((uint8_t*) internalrecvbuf, sock,
396  currSize));
397  endpointer = &internalrecvbuf[0];
398  currSize=0;
399  }
400  }
