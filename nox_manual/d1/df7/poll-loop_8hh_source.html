38 /* Represents an activity that needs periodic attention. */
39 class Pollable
40  : boost::noncopyable
41 {
42 public:
43  virtual ~Pollable() { }
44 
45  /* Called by the poll loop from any of the threads in its pool of threads.
46  * May block. If it blocks, then it may be called reentrantly from another
47  * thread, and so on if that thread blocks. It may prevent being called
48  * again by removing itself from the Poll_loop's set of Pollables with
49  * remove_pollable().
50  *
51  * While the Pollable runs, no other Pollables can run. Thus, it should
52  * limit the amount of work it can do without either blocking or returning,
53  * both of which allow other Pollables to run. Yielding with co_yield()
54  * does allow other threads to run, including other Pollables that have
55  * blocked, but it does not allow other Pollables to be polled.
56  *
57  * Should return true if some work was accomplished or if the Pollable
58  * wants to be called again without waiting, false if the Pollable had no
59  * work to do and has no need to be called again for now. */
60  virtual bool poll() = 0;
61 
62  /* Called by the poll loop when all Pollables have returned false. Should
63  * call a wait function (such as co_fd_read_wait()) that will allow the
64  * thread to wake up when there is some work to do. If there is already
65  * work to do, should call co_immediate_wake(). Must not block. */
66  virtual void wait() = 0;
67 private:
68  friend class Poll_loop_impl;
69 
70  /* Position in owning Poll_loop's list of Pollables. */
71  std::list<Pollable_ref>::iterator pollables_pos;
72 };

83  void add_pollable(Pollable* p);
84 
85  /* Remove 'p' from the set of Pollables to poll. 'p' must be currently in
86  * the Poll_loop's set of Pollables. Afterward, 'p' will not be polled or
87  * otherwise referenced again. 'p' may be currently running any number of
88  * poll operations, which may complete normally. */
89  void remove_pollable(Pollable* p);
90  void run();
91 private:
92  Poll_loop_impl* pimpl;
93 };
