215 // Internet checksum where size is number of bytes
216 //-----------------------------------------------------------------------------
217 inline
218 uint16_t
219 ip_::checksum(void * in, size_t size)
220 {
221  register size_t nleft = size;
222  const u_short *w = (u_short *) in;
223  register u_short answer;
224  register int sum = 0;
225 
226  while (nleft > 1) {
227  sum += *w++;
228  nleft -= 2;
229  }
230 
231  if (nleft == 1)
232  { sum += htons(*(u_char *) w << 8); }
233 
234  sum = (sum >> 16) + (sum & 0xffff); // add high 16 to low 16
235  sum += (sum >> 16); // add carry
236  answer = ~sum;
237  return answer;
238 }
