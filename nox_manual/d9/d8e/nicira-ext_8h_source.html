157 /* Configures the "role" of the sending controller. The default role is:
158  *
159  * - Other (NX_ROLE_OTHER), which allows the controller access to all
160  * OpenFlow features.
161  *
162  * The other possible roles are a related pair:
163  *
164  * - Master (NX_ROLE_MASTER) is equivalent to Other, except that there may
165  * be at most one Master controller at a time: when a controller
166  * configures itself as Master, any existing Master is demoted to the
167  * Slave role.
168  *
169  * - Slave (NX_ROLE_SLAVE) allows the controller read-only access to
170  * OpenFlow features. In particular attempts to modify the flow table
171  * will be rejected with an OFPBRC_EPERM error.
172  *
173  * Slave controllers also do not receive asynchronous messages
174  * (OFPT_PACKET_IN, OFPT_FLOW_REMOVED, OFPT_PORT_STATUS).
175  */
176 struct nx_role_request {
177  struct nicira_header nxh;
178  uint32_t role; /* One of NX_ROLE_*. */
179 };

190  /* Searches the flow table again, using a flow that is slightly modified
191  * from the original lookup:
192  *
193  * - The 'in_port' member of struct nx_action_resubmit is used as the
194  * flow's in_port.
195  *
196  * - If NXAST_RESUBMIT is preceded by actions that affect the flow
197  * (e.g. OFPAT_SET_VLAN_VID), then the flow is updated with the new
198  * values.
199  *
200  * Following the lookup, the original in_port is restored.
201  *
202  * If the modified flow matched in the flow table, then the corresponding
203  * actions are executed. Afterward, actions following NXAST_RESUBMIT in
204  * the original set of actions, if any, are executed; any changes made to
205  * the packet (e.g. changes to VLAN) by secondary actions persist when
206  * those actions are executed, although the original in_port is restored.
207  *
208  * NXAST_RESUBMIT may be used any number of times within a set of actions.
209  *
210  * NXAST_RESUBMIT may nest to an implementation-defined depth. Beyond this
211  * implementation-defined depth, further NXAST_RESUBMIT actions are simply
212  * ignored. (Open vSwitch 1.0.1 and earlier did not support recursion.)
213  */
214  NXAST_RESUBMIT,
215 
216  /* Set encapsulating tunnel ID. */
217  NXAST_SET_TUNNEL,
218 
219  /* Stops processing further actions, if the packet being processed is an
220  * Ethernet+IPv4 ARP packet for which the source Ethernet address inside
221  * the ARP packet differs from the source Ethernet address in the Ethernet
222  * header.
223  *
224  * This is useful because OpenFlow does not provide a way to match on the
225  * Ethernet addresses inside ARP packets, so there is no other way to drop
226  * spoofed ARPs other than sending every packet up to the controller. */
227  NXAST_DROP_SPOOFED_ARP
228 };
