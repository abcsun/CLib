60  def receive(self, recvLen=0,timeout=0):
61  """Receive command
62  If length == None, nonblocking receive (return None or message)
63  With nonblocking receive, timeout is used for select statement
64 
65  If length is zero, return single message
66  """
67  if (recvLen==0):
68  #Receive full message
69  msg=""
70  length=len(self.__header)
71  while (len(msg) < length):
72  msg+=self.sock.recv(1)
73  #Get length
74  if (len(msg) == length):
75  self.__header.unpack(msg)
76  length=self.__header.length
77  return msg
78  elif (recvLen==None):
79  #Non-blocking receive
80  ready_to_read = select.select([self.sock],[],[],timeout)[0]
81  if (ready_to_read):
82  self.__buffer += self.sock.recv(1)
83  if (len(self.__buffer) >= len(self.__header)):
84  self.__header.unpack(self.__buffer)
85  if (self.__header.length == len(self.__buffer)):
86  msg = self.__buffer
87  self.__buffer = ""
88  return msg
89  return None
90  else:
91  #Fixed length blocking receive
