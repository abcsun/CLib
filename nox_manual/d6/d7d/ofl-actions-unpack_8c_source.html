52 ofl_actions_unpack(struct ofp_action_header *src, size_t *len, struct ofl_action_header **dst, struct ofl_exp *exp) {
53 
54  if (*len < sizeof(struct ofp_action_header)) {
55  OFL_LOG_WARN(LOG_MODULE, "Received action is too short (%zu).", *len);
56  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
57  }
58 
59  if (*len < ntohs(src->len)) {
60  OFL_LOG_WARN(LOG_MODULE, "Received action has invalid length (set to %u, but only %zu received).", ntohs(src->len), *len);
61  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
62  }
63 
64  if ((ntohs(src->len) % 8) != 0) {
65  OFL_LOG_WARN(LOG_MODULE, "Received action length is not a multiple of 64 bits (%u).", ntohs(src->len));
66  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
67  }
68 
69  switch (ntohs(src->type)) {
70  case OFPAT_OUTPUT: {
71  struct ofp_action_output *sa;
72  struct ofl_action_output *da;
73 
74  if (*len < sizeof(struct ofp_action_output)) {
75  OFL_LOG_WARN(LOG_MODULE, "Received OUTPUT action has invalid length (%zu).", *len);
76  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
77  }
78 
79  sa = (struct ofp_action_output *)src;
80 
81  if (ntohl(sa->port) == 0 ||
82  (ntohl(sa->port) > OFPP_MAX && ntohl(sa->port) < OFPP_IN_PORT) ||
83  ntohl(sa->port) == OFPP_ANY) {
84  if (OFL_LOG_IS_WARN_ENABLED(LOG_MODULE)) {
85  char *ps = ofl_port_to_string(ntohl(sa->port));
86  OFL_LOG_WARN(LOG_MODULE, "Received OUTPUT action has invalid port (%s).", ps);
87  free(ps);
88  }
89  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_OUT_PORT);
90  }
91 
92  da = (struct ofl_action_output *)malloc(sizeof(struct ofl_action_output));
93  da->port = ntohl(sa->port);
94  da->max_len = ntohs(sa->max_len);
95 
96  *len -= sizeof(struct ofp_action_output);
97  *dst = (struct ofl_action_header *)da;
98  break;
99  }
100  case OFPAT_COPY_TTL_OUT: {
101  //ofp_action_header length was already checked
102  *len -= sizeof(struct ofp_action_header);
103  *dst = (struct ofl_action_header *)malloc(sizeof(struct ofl_action_header));
104  break;
105  }
106 
107  case OFPAT_COPY_TTL_IN: {
108  //ofp_action_header length was already checked
109  *len -= sizeof(struct ofp_action_header);
110  *dst = (struct ofl_action_header *)malloc(sizeof(struct ofl_action_header));
111  break;
112  }
113 
114  case OFPAT_SET_MPLS_TTL: {
115  struct ofp_action_mpls_ttl *sa;
116  struct ofl_action_mpls_ttl *da;
117 
118  if (*len < sizeof(struct ofp_action_mpls_ttl)) {
119  OFL_LOG_WARN(LOG_MODULE, "Received SET_MPLS_TTL action has invalid length (%zu).", *len);
120  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
121  }
122 
123  sa = (struct ofp_action_mpls_ttl *)src;
124 
125  da = (struct ofl_action_mpls_ttl *)malloc(sizeof(struct ofl_action_mpls_ttl));
126  da->mpls_ttl = sa->mpls_ttl;
127 
128  *len -= sizeof(struct ofp_action_mpls_ttl);
129  *dst = (struct ofl_action_header *)da;
130  break;
131  }
132 
133  case OFPAT_DEC_MPLS_TTL: {
134  //ofp_action_header length was already checked
135  *len -= sizeof(struct ofp_action_mpls_ttl);
136  *dst = (struct ofl_action_header *)malloc(sizeof(struct ofl_action_header));
137  break;
138  }
139 
140  case OFPAT_PUSH_VLAN:
141  case OFPAT_PUSH_PBB:
142  case OFPAT_PUSH_MPLS: {
143  struct ofp_action_push *sa;
144  struct ofl_action_push *da;
145

147  OFL_LOG_WARN(LOG_MODULE, "Received PUSH_VLAN/MPLS/PBB action has invalid length (%zu).", *len);
148  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);

152

154  (ntohs(sa->ethertype) != ETH_TYPE_VLAN &&
155  ntohs(sa->ethertype) != ETH_TYPE_VLAN_PBB)) ||
156  ((ntohs(src->type) == OFPAT_PUSH_MPLS) &&
157  (ntohs(sa->ethertype) != ETH_TYPE_MPLS &&
158  ntohs(sa->ethertype) != ETH_TYPE_MPLS_MCAST)) ||
159  ((ntohs(src->type) == OFPAT_PUSH_PBB) &&
160  (ntohs(sa->ethertype) != ETH_TYPE_PBB))) {
161  OFL_LOG_WARN(LOG_MODULE, "Received PUSH_VLAN/MPLS/PBB has invalid eth type. (%u)", ntohs(sa->ethertype));
162  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_ARGUMENT);
163  }
164 
165  da = (struct ofl_action_push *)malloc(sizeof(struct ofl_action_push));

167 
168  *len -= sizeof(struct ofp_action_push);

171  }

173  case OFPAT_POP_VLAN:
174  case OFPAT_POP_PBB: {
175  //ofp_action_header length was already checked
176  *len -= sizeof(struct ofp_action_header);
177  *dst = (struct ofl_action_header *)malloc(sizeof(struct ofl_action_header));
178  break;
179  }

259  da->nw_ttl = sa->nw_ttl;

266  case OFPAT_DEC_NW_TTL: {
267  //ofp_action_header length was already checked
268  *len -= sizeof(struct ofp_action_header);
269  *dst = (struct ofl_action_header *)malloc(sizeof(struct ofl_action_header));
270  break;
271  }
272 
273  case OFPAT_SET_FIELD: {
274  struct ofp_action_set_field *sa;
275  struct ofl_action_set_field *da;
276  uint8_t *value;
277 
278  sa = (struct ofp_action_set_field*) src;
279  da = (struct ofl_action_set_field *)malloc(sizeof(struct ofl_action_set_field));
280  da->field = (struct ofl_match_tlv*) malloc(sizeof(struct ofl_match_tlv));
281 
282  memcpy(&da->field->header,sa->field,4);
283  da->field->header = ntohl(da->field->header);
284  value = (uint8_t *) src + sizeof (struct ofp_action_set_field);
285  da->field->value = (uint8_t*) malloc(OXM_LENGTH(da->field->header));
286  switch(OXM_LENGTH(da->field->header)){
287  case 1:
288  case 6:
289  case 16:
290  memcpy(da->field->value , value, OXM_LENGTH(da->field->header));
291  break;
292 
293  case 2:{
294  uint16_t v = ntohs(*((uint16_t*) value));
295  memcpy(da->field->value , &v, OXM_LENGTH(da->field->header));
296  break;
297  }
298  case 4:{
299  uint32_t v;
300  uint8_t field = OXM_FIELD(da->field->header);
301  if( field != 11 && field != 12 && field != 22 && field != 23)
302  v = htonl(*((uint32_t*) value));
303  else v = *((uint32_t*) value);
304  memcpy(da->field->value , &v, OXM_LENGTH(da->field->header));
305  break;
306  }
307  case 8:{
308  uint64_t v = hton64(*((uint64_t*) value));
309  memcpy(da->field->value , &v, OXM_LENGTH(da->field->header));
310  break;
311  }
312  }
313  *len -= ROUND_UP(ntohs(src->len),8);
314  *dst = (struct ofl_action_header *)da;
315  break;
316  }
317 
318  case OFPAT_EXPERIMENTER: {
319  ofl_err error;
320 
321  if (*len < sizeof(struct ofp_action_experimenter_header)) {
322  OFL_LOG_WARN(LOG_MODULE, "Received EXPERIMENTER action has invalid length (%zu).", *len);
323  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
324  }
325 
326  if (exp == NULL || exp->act == NULL || exp->act->unpack == NULL) {
327  OFL_LOG_WARN(LOG_MODULE, "Received EXPERIMENTER action, but no callback is given.");
328  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_EXPERIMENTER);
329  }
330  error = exp->act->unpack(src, len, dst);
331  if (error) {
332  return error;
333  }
334  break;
335  }
336 
337  default: {
338  OFL_LOG_WARN(LOG_MODULE, "Received unknown action type (%u).", ntohs(src->type));
339  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_TYPE);
340  }
341  }
342  (*dst)->type = (enum ofp_action_type)ntohs(src->type);
343 
344  return 0;
