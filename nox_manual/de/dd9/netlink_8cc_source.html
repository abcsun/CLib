40 /* Netlink PID.
41  *
42  * Every Netlink socket must be bound to a unique 32-bit PID. By convention,
43  * programs that have a single Netlink socket use their Unix process ID as PID,
44  * and programs with multiple Netlink sockets add a unique per-socket
45  * identifier in the bits above the Unix process ID.
46  *
47  * The kernel has Netlink PID 0.
48  */
49 namespace {
50 
51 /* Parameters for how many bits in the PID should come from the Unix process ID
52  * and how many unique per-socket. */
53 const int socket_bits = 10;
54 const int process_bits = 22;
55 const uint32_t process_mask = (1u << process_bits) - 1;
56 BOOST_STATIC_ASSERT(socket_bits + process_bits == 32);
57 
58 /* Bit vector of unused socket identifiers. */
59 std::vector<bool> avail_sockets(1u << socket_bits, 0);
60 
61 /* Allocates and returns a new Netlink PID. */
62 uint32_t alloc_pid()
63 {
64  std::vector<bool>& bm = avail_sockets;
65  std::vector<bool>::iterator i = std::find(bm.begin(), bm.end(), 0);
66  if (i == bm.end()) {
67  throw std::runtime_error("out of sockets in allocate_pid");
68  }
69  *i = 1;
70  return (::getpid () & process_mask) | ((i - bm.begin()) << process_bits);
71 }
72 
73 /* Makes the specified 'pid' available for reuse. */
74 void free_pid(uint32_t pid)
75 {
76  assert(avail_sockets[pid >> process_bits]);
77  avail_sockets[pid >> process_bits] = 0;
78 }

131 /* Binds the Netlink socket according to 'p', and connects the remote address
132  * as the kernel.
133  * Returns 0 if successful, otherwise a Unix error number.
134  *
135  * A bound Netlink socket may be rebound at any time if desired. */
136 int Nl_socket::bind(const Parameters& p)
137 {
138  fd.reset(::socket(AF_NETLINK, SOCK_RAW, p.protocol));
139  if (fd < 0) {
140  return errno;
141  }
142 
143  int flags = fcntl(fd, F_GETFL, 0);
144  if (flags < 0 || fcntl(fd, F_SETFL, flags | O_NONBLOCK) < 0) {
145  log.err("fcntl: %s", strerror(errno));
146  return errno;
147  }
148 
149  if (p.send_buffer != 0
150  && ::setsockopt(fd, SOL_SOCKET, SO_SNDBUF,
151  &p.send_buffer, sizeof p.send_buffer) < 0) {
152  log.err("setsockopt(SO_SNDBUF,%zu): %s",
153  p.send_buffer, strerror(errno));
154  return errno;
155  }
156 
157  if (p.receive_buffer != 0
158  && ::setsockopt(fd, SOL_SOCKET, SO_RCVBUF,
159  &p.receive_buffer, sizeof p.receive_buffer) < 0) {
160  log.err("setsockopt(SO_RCVBUF,%zu): %s",
161  p.send_buffer, strerror(errno));
162  return errno;
163  }
164 
165  /* Bind local address as our selected pid. */
166  sockaddr_nl local;
167  memset(&local, 0, sizeof local);
168  local.nl_family = AF_NETLINK;
169  local.nl_pid = m_pid;
170  if (p.group > 0 && p.group <= 32) {
171  local.nl_groups |= 1ul << (p.group - 1);
172  }
173  if (::bind(fd, (sockaddr*) &local, sizeof local) < 0) {
174  log.err("bind(%"PRIu32"): %s", m_pid, strerror(errno));
175  return errno;
176  }
177 
178  /* Bind remote address as the kernel (pid 0). */
179  sockaddr_nl remote;
180  memset(&remote, 0, sizeof remote);
181  remote.nl_family = AF_NETLINK;
182  remote.nl_pid = 0;
183  if (::connect(fd, (sockaddr*) &remote, sizeof remote) < 0) {
184  log.err("connect(0): %s", strerror(errno));
185  return errno;
186  }
187 
188  if (p.group > 32
189  && ::setsockopt(fd, SOL_NETLINK, NETLINK_ADD_MEMBERSHIP,
190  &p.group, sizeof p.group) < 0) {
191  log.err("setsockopt(NETLINK_ADD_MEMBERSHIP,%d): %s",
192  p.group, strerror(errno));
193  return errno;
194  }
195 
196  return 0;
197 }

215 /* Calls ::recv on fd with the specified arguments, with basic handling for
216  * error conditions, and passes along ::recv's return value. */
217 ssize_t
218 Nl_socket::call_recv(void* buf, size_t nbytes, int flags)
219 {
220  for (;;) {
221  ssize_t retval = ::recv(fd, buf, nbytes, flags);
222  if (retval < 0) {
223  if (errno == ENOBUFS) {
224  log.warn("lost packet (high data rate?)");
225  retransmit_all();
226  continue;
227  } else {
228  if (errno != EAGAIN) {
229  log.err("recv: %s", strerror(errno));
230  }
231  return -errno;
232  }
233  }
234  return retval;
235  }

238 std::auto_ptr<Buffer>
239 Nl_socket::do_recv(int& error)
240 {
241  for (;;) {
242  /* First find out the length of the message. */
243  uint8_t tmp;
244  error = call_recv(&tmp, 1, MSG_PEEK | MSG_TRUNC);
245  if (error <= 0) {
246  error = -error;
247  return std::auto_ptr<Buffer>(0);
248  }
249 
250  /* Then read the whole message. */
251  std::auto_ptr<Buffer> buffer(new Array_buffer(error));
252  error = call_recv(buffer->data(), error, 0);
253  if (error < 0) {
254  error = -error;
255  return std::auto_ptr<Buffer>(0);
256  }
257  buffer->trim(error);
258 
259  Nl_message msg(buffer);
260  if (!msg.ok()) {
261  log.warn("received invalid Netlink message");
262  error = EPROTO;
263  return std::auto_ptr<Buffer>(0);
264  }
265 
266  uint32_t seq = msg.nlhdr().nlmsg_seq;
267  Transaction_map::iterator i(transactions.find(seq));
268  if (i == transactions.end()) {
269  error = 0;
270  return msg.steal();
271  }
272 
273  Nl_transaction& t = *i->second;
274  if (t.status < 0) {
275  t.reply = msg.steal();
276  t.status = 0;
277  t.wakeup.broadcast();
278  }
279  }

312 /* Sends 'msg' to the kernel and waits for a response.
313  *
314  * This function is reliable even though Netlink is an unreliable datagram
315  * transport. The message send to the kernel is reliable: the kernel will tell
316  * us if the message cannot be queued (and we will in that case put it on the
317  * transmit queue and wait until it can be delivered). Receiving the reply is
318  * the real problem: if the socket buffer is full when the kernel tries to send
319  * the reply, the reply will be dropped. However, the kernel lets us know that
320  * a reply was dropped by returning ENOBUFS from ::recv. We deal with this by
321  * resending all outstanding requests.
322  *
323  * This retransmission mechanism only functions properly for idempotent
324  * requests.
325  *
326  * It's best not to mix transactional and non-transactional messages on a
327  * single socket, as non-transactional data can get dropped even *after* it is
328  * received because we're waiting for the reply to a transaction.
329  */
330 std::auto_ptr<Buffer>
331 Nl_socket::do_transact(Nl_transaction& t, int& error)
332 {
333  for (;;) {
334  switch (t.status) {
335  case Nl_transaction::NEED_RETRANSMIT:
336  error = t.retransmit();
337  if (error) {
338  return std::auto_ptr<Buffer>(0);
339  }
340  break;
341 
342  case Nl_transaction::NEED_REPLY:
343  /* Receive a message.
344  *
345  * If the message is not part of a transaction, it will get thrown
346  * away. We could queue a few received non-transactional messages,
347  * but we'd still have to start throwing them away after a while.
348  * It's probably a better idea to use separate Netlink sockets for
349  * transactional and non-transactional traffic.
350  *
351  * Also, we can get duplicate responses to transactional messages,
352  * in the case where we retransmit a request that doesn't need to
353  * be, and those shouldn't be queued at all. */
354  do_recv(error);
355  if (error) {
356  return std::auto_ptr<Buffer>(0);
357  }
358  break;
359 
360  case 0: {
361  Nl_message m(t.reply);
362  if (!m.ok()) {
363  log.err("Nl_message too short");
364  error = EPROTO;
365  return std::auto_ptr<Buffer>(0);
366  }
367 
368  if (m.nlhdr().nlmsg_type == NLMSG_ERROR) {
369  const nlmsgerr* nlerr = m.pull<nlmsgerr>();
370  if (!nlerr) {
371  log.err("NLMSG_ERROR packet too short");
372  error = EPROTO;
373  return std::auto_ptr<Buffer>(0);
374  } else if (nlerr->error < 0) {
375  error = -nlerr->error;
376  if (error == EAGAIN) {
377  error = EPROTO;
378  }
379  log.err("NLMSG_ERROR received from kernel: %d (%s)",
380  error, strerror(error));
381  return std::auto_ptr<Buffer>(0);
382  }
383  }
384 
385  error = 0;
386  return m.steal();
387  }
388 
389  default:
390  assert(t.status > 0);
391  error = t.status;
392  return std::auto_ptr<Buffer>(0);
393  }
394  }

397 void
398 Nl_socket::retransmit_all()
399 {
400  BOOST_FOREACH (Transaction_map::value_type& i, transactions) {
401  Nl_transaction& t = *i.second;
402  if (t.status == Nl_transaction::NEED_REPLY) {
403  t.status = Nl_transaction::NEED_RETRANSMIT;
404  t.retransmit();
405 
406  /* Transaction may be waiting to receive a message, but now it
407  * needs to wait to send a message, so wake it up. */
408  t.wakeup.broadcast();
409  } else {
410  /* Transaction is already waiting to retransmit. Nothing to do. */
411  }
412  }
413 }

415 void
416 Nl_socket::transact_wait(Nl_transaction& t)
417 {
418  /* Need to retransmit? */
419  int error = t.retransmit();
420  if (error) {
421  if (error == EAGAIN) {
422  co_fd_write_wait(fd, NULL);
423  } else {
424  co_immediate_wake(1, NULL);
425  }
426  return;
427  }
428 
429  /* Need to receive? */
430  if (t.status == Nl_transaction::NEED_REPLY) {
431  /* Need to wait for messages to arrive. */
432  co_fd_read_wait(fd, NULL);
433 
434  /* Also need to wait for someone else to get our message and
435  * complete the transaction for us. We won't necessarily be
436  * awakened by the FD becoming ready, because another thread can
437  * read from the FD without ever coming through the poll loop. */
438  t.wakeup.wait();
439  return;
440  }
441 
442  /* Transaction complete! */
443  co_immediate_wake(1, NULL);
444 }

472 Nl_transaction::retransmit()
473 {
474  int error = 0;
475  if (status == NEED_RETRANSMIT) {
476  error = socket.send(request.all(), false);
477  if (!error) {
478  status = NEED_REPLY;
479  } else if (error != EAGAIN) {

482  }

486 /* Returns a new sequence number for sending a request.
487  *
488  * This implementation uses sequence numbers that are unique process-wide, to
489  * avoid a hypothetical race: send request, close socket, open new socket that
490  * reuses the old socket's PID value, send request on new socket, receive reply
491  * from kernel to old socket but with same PID and sequence number. This race
492  * could be avoided other ways, so process-wide uniqueness is not a guarantee

512 /* Constructs an Nl_message for composing a Netlink message with the given
513  * 'type' and 'flags'. 'sock' is used to obtain a PID and sequence number for
514  * proper routing of replies. 'expected_payload' may be specified as an
515  * estimate of the number of payload bytes to be supplied; if the size of the
516  * payload is unknown the default of 0 is acceptable.
517  *
518  * 'type' is ordinarily an enumerated value specific to the Netlink protocol
519  * (e.g. RTM_NEWLINK, for NETLINK_ROUTE protocol). For Generic Netlink, 'type'
520  * is the family number obtained via Genl_family::number().
521  *
522  * 'flags' is a bit-mask that indicates what kind of request is being made. It
523  * is often NLM_F_REQUEST indicating that a request is being made, commonly
524  * or'd with NLM_F_ACK to request an acknowledgement.
525  *
526  * Genl_message::Genl_message() provides a more convenient interface for
527  * composing a Generic Netlink message. */
528 Nl_message::Nl_message(uint32_t type, uint32_t flags, Nl_socket& sock,
529  size_t expected_payload)
530 {
531  size_t size = NLMSG_HDRLEN + NLMSG_ALIGN(expected_payload);
532  buffer.reset(new Array_buffer(size));
533  buffer->trim(0);
534  offset = 0;
535 
536  nlmsghdr* h = put<nlmsghdr>();
537  h->nlmsg_len = buffer->size();
538  h->nlmsg_type = type;
539  h->nlmsg_flags = flags;
540  h->nlmsg_seq = sock.next_seq();
541  h->nlmsg_pid = sock.pid();
542 }

568 static void nlmsghdr_to_string(const nlmsghdr& h, char *p)
569 {
570  sprintf(p, "nl(len:%"PRIu32", type=%"PRIu16,
571  h.nlmsg_len, h.nlmsg_type);
572  if (h.nlmsg_type == NLMSG_NOOP) {
573  strcat(p, "(no-op)");
574  } else if (h.nlmsg_type == NLMSG_ERROR) {
575  strcat(p, "(error)");
576  } else if (h.nlmsg_type == NLMSG_DONE) {
577  strcat(p, "(done)");
578  } else if (h.nlmsg_type == NLMSG_OVERRUN) {
579  strcat(p, "(overrun)");
580  } else if (h.nlmsg_type < NLMSG_MIN_TYPE) {
581  strcat(p, "(reserved)");
582  } else {
583  strcat(p, "(family-defined)");
584  }
585  sprintf(strchr(p, '\0'), ", flags=%"PRIu16, h.nlmsg_flags);
586  if (h.nlmsg_flags & NLM_F_REQUEST) {
587  strcat(p, "[REQUEST]");
588  }
589  if (h.nlmsg_flags & NLM_F_MULTI) {
590  strcat(p, "[MULTI]");
591  }
592  if (h.nlmsg_flags & NLM_F_ACK) {
593  strcat(p, "[ACK]");
594  }
595  if (h.nlmsg_flags & NLM_F_ECHO) {
596  strcat(p, "[ECHO]");
597  }
598  if (h.nlmsg_flags & ~(NLM_F_REQUEST | NLM_F_MULTI
599  | NLM_F_ACK | NLM_F_ECHO)) {
600  strcat(p, "[OTHER]");
601  }
602  if (h.nlmsg_flags == 0) {
603  strcat(p, "0");
604  }
605  sprintf(strchr(p, '\0'), ", seq=%"PRIx32", pid=%"PRIu32"(%d:%d))",
606  h.nlmsg_seq, h.nlmsg_pid, (int) (h.nlmsg_pid & process_mask),
607  (int) (h.nlmsg_pid >> process_bits));
608 }

610 std::string Nl_message::to_string() const
611 {
612  if (buffer->size() < NLMSG_HDRLEN) {
613  return "nl(truncated)";
614  } else {
615  char buf[4096];
616  nlmsghdr& h = nlhdr();
617  nlmsghdr_to_string(h, buf);
618  if (h.nlmsg_type == NLMSG_ERROR) {
619  if (all().size()
620  < NLMSG_HDRLEN + NLMSG_ALIGN(sizeof(struct nlmsgerr))) {
621  strcat(buf, " error(truncated)");
622  } else {
623  const nlmsgerr& e = all().at<nlmsgerr>(NLMSG_HDRLEN);
624  sprintf(strchr(buf, '\0'), " error(%d", e.error);
625  if (e.error < 0) {
626  sprintf(strchr(buf, '\0'), "(%s)", strerror(-e.error));
627  }
628  strcat(buf, ", in-reply-to(");
629  nlmsghdr_to_string(e.msg, strchr(buf, '\0'));
630  strcat(buf, "))");
631  }
632  }
633  return buf;
634  }

837 void Nl_policy::init(Nl_attr elements_[], size_t n)
838 {

854 /* Implementation of template<> Nl_policy::parse. */
855 bool Nl_policy::parse(const Buffer& b_, nlattr* attrs[], size_t n)
856 {
857  assert(n >= elements.size());
858  for (size_t i = 0; i < n; i++) {
859  attrs[i] = NULL;
860  }
861 
862  size_t still_required = n_required;
863  Nonowning_buffer b = b_;
864  while (b.size() > 0) {
865  nlattr* nla = b.try_pull<nlattr>();
866  if (!nla) {
867  log.warn("attribute shorter than NLA_HDRLEN (%zu)",
868  b.size());
869  return false;
870  }
871 
872  /* Make sure its claimed length is plausible. */
873  if (nla->nla_len < NLA_HDRLEN) {
874  log.warn("attribute shorter than NLA_HDRLEN (%"PRIu16")",
875  nla->nla_len);
876  return false;
877  }
878  size_t len = nla->nla_len - NLA_HDRLEN;
879  size_t aligned_len = NLA_ALIGN(len);
880  if (aligned_len > b.size()) {
881  log.warn("attribute %"PRIu16" aligned data len (%zu) "
882  "> bytes left (%zu)",
883  nla->nla_type, aligned_len, b.size());
884  return false;
885  }
886 
887  uint16_t type = nla->nla_type;
888  if (type < elements.size() && elements[type] != NULL) {
889  const Nl_attr& e = *elements[type];
890 
891  /* Validate length and content. */
892  if (len < e.min_len || len > e.max_len) {
893  log.warn("attribute %"PRIu16" length %zu not in allowed range "
894  "%zu...%zu", type, len, e.min_len, e.max_len);
895  return false;
896  }
897  if (e.value_type == Nl_attr::NUL_STRING
898  && ((char*) nla)[nla->nla_len - 1] != '\0') {
899  log.warn("attribute %"PRIu16" lacks null terminator", type);
900  return false;
901  }
902 
903  if (e.required && attrs[type] == NULL) {
904  --still_required;
905  }
906  attrs[type] = nla;
907  } else {
908  /* Skip attribute type that we don't care about. */
909  }
910 
911  b.pull(aligned_len);
912  }
913  if (still_required) {
914  log.warn("%zu required attributes missing", still_required);
915  return false;
916  }
917  return true;
918 }
