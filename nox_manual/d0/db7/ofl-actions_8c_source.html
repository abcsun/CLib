44  switch (act->type) {

50  return;

52  }
53  case OFPAT_OUTPUT:
54  case OFPAT_COPY_TTL_OUT:
55  case OFPAT_COPY_TTL_IN:

71  if (exp == NULL || exp->act == NULL || exp->act->free == NULL) {
72  OFL_LOG_WARN(LOG_MODULE, "Freeing experimenter action, but no callback is given.");

84 ofl_err
85 ofl_utils_count_ofp_actions(void *data, size_t data_len, size_t *count) {
86  struct ofp_action_header *act;
87  uint8_t *d;
88 
89  d = (uint8_t *)data;
90  *count = 0;
91 
92  /* this is needed so that buckets are handled correctly */
93  while (data_len >= sizeof(struct ofp_action_header) -4 ) {
94  act = (struct ofp_action_header *)d;
95  if (data_len < ntohs(act->len) || ntohs(act->len) < sizeof(struct ofp_action_header) - 4) {
96  OFL_LOG_WARN(LOG_MODULE, "Received action has invalid length.");
97  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
98  }
99  data_len -= ntohs(act->len);
100  d += ntohs(act->len);
101  (*count)++;
102  }
103 
104  return 0;
105 }
