67 ofl_exp_msg_unpack(struct ofp_header *oh, size_t *len, struct ofl_msg_experimenter **msg) {
68  struct ofp_experimenter_header *exp;
69 
70  if (*len < sizeof(struct ofp_experimenter_header)) {
71  OFL_LOG_WARN(LOG_MODULE, "Received EXPERIMENTER message is shorter than ofp_experimenter_header.");
72  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
73  }
74 
75  exp = (struct ofp_experimenter_header *)oh;
76 
77  switch (htonl(exp->experimenter)) {
78  case (OPENFLOW_VENDOR_ID): {
79  return ofl_exp_openflow_msg_unpack(oh, len, msg);
80  }
81  case (NX_VENDOR_ID): {
82  return ofl_exp_nicira_msg_unpack(oh, len, msg);
83  }
84  default: {
85  OFL_LOG_WARN(LOG_MODULE, "Trying to unpack unknown EXPERIMENTER message (%u).", htonl(exp->experimenter));
86  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_EXPERIMENTER);
87  }
88  }
89 }
