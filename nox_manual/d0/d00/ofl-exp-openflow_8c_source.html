47 ofl_exp_openflow_msg_pack(struct ofl_msg_experimenter *msg, uint8_t **buf, size_t *buf_len) {
48  if (msg->experimenter_id == OPENFLOW_VENDOR_ID) {
49  struct ofl_exp_openflow_msg_header *exp = (struct ofl_exp_openflow_msg_header *)msg;
50  switch (exp->type) {
51  case (OFP_EXT_QUEUE_MODIFY):
52  case (OFP_EXT_QUEUE_DELETE): {
53  struct ofl_exp_openflow_msg_queue *q = (struct ofl_exp_openflow_msg_queue *)exp;
54  struct openflow_queue_command_header *ofp;
55 
56  *buf_len = sizeof(struct openflow_queue_command_header) + ofl_structs_packet_queue_ofp_len(q->queue);
57  *buf = (uint8_t *)malloc(*buf_len);
58 
59  ofp = (struct openflow_queue_command_header *)(*buf);
60  ofp->header.vendor = htonl(exp->header.experimenter_id);
61  ofp->header.subtype = htonl(exp->type);
62  ofp->port = htonl(q->port_id);
63 
64  ofl_structs_packet_queue_pack(q->queue, (struct ofp_packet_queue *)ofp->body);
65  return 0;
66  }
67  case (OFP_EXT_SET_DESC): {
68  struct ofl_exp_openflow_msg_set_dp_desc *s = (struct ofl_exp_openflow_msg_set_dp_desc *)exp;
69  struct openflow_ext_set_dp_desc *ofp;
70 
71  *buf_len = sizeof(struct openflow_ext_set_dp_desc);
72  *buf = (uint8_t *)malloc(*buf_len);
73 
74  ofp = (struct openflow_ext_set_dp_desc *)(*buf);
75  ofp->header.vendor = htonl(exp->header.experimenter_id);
76  ofp->header.subtype = htonl(exp->type);
77  strncpy(ofp->dp_desc, s->dp_desc, DESC_STR_LEN);
78 
79  return 0;
80  }
81  default: {
82  OFL_LOG_WARN(LOG_MODULE, "Trying to print unknown Openflow Experimenter message.");
83  return -1;
84  }
85  }
86  } else {
87  OFL_LOG_WARN(LOG_MODULE, "Trying to print non-Openflow Experimenter message.");
88  return -1;
89  }
90 }

92 ofl_err
93 ofl_exp_openflow_msg_unpack(struct ofp_header *oh, size_t *len, struct ofl_msg_experimenter **msg) {
94  struct ofp_extension_header *exp;
95 
96  if (*len < sizeof(struct ofp_extension_header)) {
97  OFL_LOG_WARN(LOG_MODULE, "Received EXPERIMENTER message has invalid length (%zu).", *len);
98  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
99  }
100 
101  exp = (struct ofp_extension_header *)oh;
102 
103  if (ntohl(exp->vendor) == OPENFLOW_VENDOR_ID) {
104 
105  switch (ntohl(exp->subtype)) {
106  case (OFP_EXT_QUEUE_MODIFY):
107  case (OFP_EXT_QUEUE_DELETE): {
108  struct openflow_queue_command_header *src;
109  struct ofl_exp_openflow_msg_queue *dst;
110  ofl_err error;
111 
112  if (*len < sizeof(struct openflow_queue_command_header)) {
113  OFL_LOG_WARN(LOG_MODULE, "Received EXT_QUEUE_MODIFY message has invalid length (%zu).", *len);
114  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
115  }
116  *len -= sizeof(struct openflow_queue_command_header);
117 
118  src = (struct openflow_queue_command_header *)exp;
119 
120  dst = (struct ofl_exp_openflow_msg_queue *)malloc(sizeof(struct ofl_exp_openflow_msg_queue));
121  dst->header.header.experimenter_id = ntohl(exp->vendor);
122  dst->header.type = ntohl(exp->subtype);
123  dst->port_id = ntohl(src->port);
124 
125  error = ofl_structs_packet_queue_unpack((struct ofp_packet_queue *)(src->body), len, &(dst->queue));
126  if (error) {
127  free(dst);
128  return error;
129  }
130 
131  (*msg) = (struct ofl_msg_experimenter *)dst;
132  return 0;
133  }
134  case (OFP_EXT_SET_DESC): {
135  struct openflow_ext_set_dp_desc *src;
136  struct ofl_exp_openflow_msg_set_dp_desc *dst;
137 
138  if (*len < sizeof(struct openflow_ext_set_dp_desc)) {
139  OFL_LOG_WARN(LOG_MODULE, "Received EXT_SET_DESC message has invalid length (%zu).", *len);
140  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
141  }
142  *len -= sizeof(struct openflow_ext_set_dp_desc);
143 
144  src = (struct openflow_ext_set_dp_desc *)exp;
145 
146  dst = (struct ofl_exp_openflow_msg_set_dp_desc *)malloc(sizeof(struct ofl_exp_openflow_msg_set_dp_desc));
147  dst->header.header.experimenter_id = ntohl(exp->vendor);
148  dst->header.type = ntohl(exp->subtype);
149 
150  dst->dp_desc = strcpy((char *)malloc(strlen(src->dp_desc)+1), src->dp_desc);
151 
152  (*msg) = (struct ofl_msg_experimenter *)dst;
153  return 0;
154  }
155  default: {
156  OFL_LOG_WARN(LOG_MODULE, "Trying to unpack unknown Openflow Experimenter message.");
157  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_EXPERIMENTER);
158  }
159  }
160  } else {
161  OFL_LOG_WARN(LOG_MODULE, "Trying to unpack non-Openflow Experimenter message.");
162  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_EXPERIMENTER);
163  }
164  free(msg);
165  return 0;
166 }

172  switch (exp->type) {

186  }
187  }

197  char *str;

199  FILE *stream = open_memstream(&str, &str_size);
200 
201  if (msg->experimenter_id == OPENFLOW_VENDOR_ID) {
202  struct ofl_exp_openflow_msg_header *exp = (struct ofl_exp_openflow_msg_header *)msg;
203  switch (exp->type) {
204  case (OFP_EXT_QUEUE_MODIFY):
205  case (OFP_EXT_QUEUE_DELETE): {
206  struct ofl_exp_openflow_msg_queue *q = (struct ofl_exp_openflow_msg_queue *)exp;
207  fprintf(stream, "%squeue{port=\"", exp->type == OFP_EXT_QUEUE_MODIFY ? "mod" : "del");
208  ofl_port_print(stream, q->port_id);
209  fprintf(stream, "\", queue=");
210  ofl_structs_queue_print(stream, q->queue);
211  fprintf(stream, "}");
212  break;
213  }
214  case (OFP_EXT_SET_DESC): {
215  struct ofl_exp_openflow_msg_set_dp_desc *s = (struct ofl_exp_openflow_msg_set_dp_desc *)exp;
216  fprintf(stream, "setdesc{desc=\"%s\"}", s->dp_desc);
217  break;
218  }
219  default: {
220  OFL_LOG_WARN(LOG_MODULE, "Trying to print unknown Openflow Experimenter message.");
221  fprintf(stream, "ofexp{type=\"%u\"}", exp->type);
222  }
223  }
224  } else {
225  OFL_LOG_WARN(LOG_MODULE, "Trying to print non-Openflow Experimenter message.");
226  fprintf(stream, "exp{exp_id=\"%u\"}", msg->experimenter_id);
227  }
228 
229  fclose(stream);
230  return str;
