183 {

185  Name_to_module::iterator i = pimpl->name_to_module.find(short_name);
186  if (i == pimpl->name_to_module.end()) {

192  Module module = pimpl->module_to_name.size();
193  pimpl->module_to_name.push_back(short_name);
194  Name_to_module::value_type value(short_name, module);
195  i = pimpl->name_to_module.insert(value).first;
196 
197  /* Set log levels. */
198  for (Facility facility = 0; facility < N_FACILITIES; ++facility) {
199  pimpl->levels[facility].push_back(pimpl->default_levels[facility]);
200  }
201  }
202  return i->second;

317 //-----------------------------------------------------------------------------
318 // Set debugging levels:
319 //
320 // mod:facility:level mod2:facility:level
321 std::string
322 Vlog::set_levels_from_string(const std::string& str)
323 {
324  using namespace std;
325  using namespace boost;
326 
327  char_separator<char> sepwhite(" \t");
328  char_separator<char> sepcol(":");
329 
330  typedef tokenizer<char_separator<char> > charseptok;
331 
332  charseptok tok(str,sepwhite);
333 
334  for(charseptok::iterator beg = tok.begin(); beg != tok.end(); ++beg) {
335  charseptok curmod(*beg, sepcol);
336 
337  /* Parse module. */
338  charseptok::iterator moditer = curmod.begin();
339  Module module;
340  if (*moditer == "ANY") {
341  module = ANY_MODULE;
342  } else {
343  module = get_module_val(moditer->c_str(), true);
344  if (module == -1) {
345  return "unknown module " + *moditer;
346  }
347  }
348 
349  /* Parse facility. */
350  iter_inc_throw_on_end(curmod, moditer, "invalid token "+*beg, false);
351  Facility facility;
352  if (*moditer == "ANY") {
353  facility = ANY_FACILITY;
354  } else {
355  facility = get_facility_val((*moditer).c_str());
356  if (facility == -1) {
357  return "unknown facility " + *moditer;
358  }
359  }
360 
361  /* Parse level. */
362  iter_inc_throw_on_end(curmod, moditer, "invalid token "+*beg, false);
363  Level level = get_level_val((*moditer).c_str());
364  if (level == -1) {
365  return "unknown level " + *moditer;
366  }
367  iter_inc_throw_on_end(curmod, moditer, "invalid token "+*beg, true);
368 
369  /* Set level. */
370  set_levels(facility, module, level);
371  }
372  return "ack";
373 }

392 void
393 Vlog::output(Module module, Level level, const char* log_msg)
394 {
395  pimpl->msg_num++;
396 
397  int save_errno = errno;
398 
399  const char* module_name = get_module_name(module);
400  const char* level_name = get_level_name(level);
401  if (pimpl->levels[FACILITY_CONSOLE][module] >= level) {
402  size_t length = strlen(log_msg);
403  bool needs_new_line = !length || log_msg[length - 1] != '\n';
404  ::fprintf(stderr, "%05d|%s|%s:%s%s",
405  pimpl->msg_num, module_name, level_name, log_msg,
406  needs_new_line ? "\n" : "");
407  }
408 
409  if (pimpl->levels[FACILITY_SYSLOG][module] >= level) {
410  int priority
411  = (level == LEVEL_EMER ? LOG_EMERG
412  : level == LEVEL_ERR ? LOG_ERR
413  : level == LEVEL_WARN ? LOG_WARNING
414  : level == LEVEL_INFO ? LOG_INFO
415  : LOG_DEBUG);
416  if(strlen(log_msg) < MAX_MSG_LEN) {
417  ::syslog(priority, "%05d|%s:%s %s",
418  pimpl->msg_num, module_name, level_name, log_msg);
419  } else {
420  // this is a long message, so we will need to split it into multiple calls
421  // to syslog, each of length <= MAX_MSG_LEN
422  // Do this in a separate branch to avoid copies on short log messages
423  std::string long_str = std::string(log_msg);
424  int start_index = 0;
425  while(start_index < long_str.length()) {
426  std::string sub = long_str.substr(start_index, MAX_MSG_LEN);
427  ::syslog(priority, "%05d|%s:%s %s",
428  pimpl->msg_num, module_name, level_name, sub.c_str());
429  start_index += MAX_MSG_LEN;
430  }
431  }
432  }
433 /*
434  // Send log msg to socket
435  char pWrite[MAX_MSG_LEN];
436  snprintf(pWrite, MAX_MSG_LEN, "%05d|%s|%s:%s\n",pimpl->msg_num,module_name,level_name,log_msg);
437  sendto(hSock,pWrite,strlen(pWrite),0,(sockaddr*)&addr,sizeof(addr));
438 */
439  /* Restore errno (it's pretty unfriendly for a log function to change
440  * errno). */
441  errno = save_errno;
442 }
