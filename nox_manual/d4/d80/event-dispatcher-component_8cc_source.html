39 EventDispatcherComponent::EventDispatcherComponent(const Context* c,
40  const json_object*
41  platformconf)
42 
43  : Component(c) {
44  // Construct the 'filter sequences' defined the JSON configuration file
45  json_object* nox = json::get_dict_value(platformconf, "nox");
46  json_object* events = json::get_dict_value(nox, "events");
47  json_dict* eventsDict = (json_dict*) events->object;
48 
49  // For every event defined in the configuration
50  json_dict::iterator event_di;
51  for(event_di=eventsDict->begin(); event_di!=eventsDict->end(); ++event_di) {
52  string event_name = event_di->first;
53  int order = 0;
54  json_array::iterator li;
55  json_array* compList = (json_array*) event_di->second->object;
56  // for every filter under the event
57  EventFilterChain chain;
58  for(li=compList->begin(); li!=compList->end(); ++li) {
59  // add the filter in the event's filter_chain
60  string filter = (((json_object*)*li)->get_string(true));
61  chain[filter] = order++;
62  }
63  filter_chains[event_name] = chain;
64  }
65 
66  // Register the system events
67  register_event<Datapath_join_event>();
68  register_event<Datapath_leave_event>();
69  register_event<Shutdown_event>();
70  register_event<Bootstrap_complete_event>();

142 bool
143 EventDispatcherComponent::register_handler(const Component_name& filter,
144  const Event_name& name,
145  const Event_handler& h) const {
146  if (events.find(name) == events.end()) {
147  return false;
148  }
149 
150  if (filter_chains.find(name) == filter_chains.end()) {
151  nox::register_handler(name, h, 0);
152  } else {
153  EventFilterChain& chain = filter_chains[name];
154  if (chain.find(filter) == chain.end()) {
155  nox::register_handler(name, h, 0);
156  } else {
157  nox::register_handler(name, h, chain[filter]);
158  }
159  }
160 
161  return true;
162 }
