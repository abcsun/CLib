113 Application_list
114 parse_application_list(int optind, int argc, char** argv) {
115  Application_list applications;
116 
117  set<string> applications_defined;
118  for (int i = optind; i < argc; i++) {
119  vector<string> tokens;
120  for (char* token = strtok(argv[i], "="); token != NULL;
121  token = strtok(NULL, ",")) {
122  tokens.push_back(token);
123  }
124  if (tokens.empty()) {
125  lg.err("Empty argument on command line");
126  continue;
127  }
128 
129  string app = tokens.front();
130  if (applications_defined.find(app) != applications_defined.end()) {
131  lg.err("Application '%s' defined multiple times in the command line. "
132  "Ignoring the latter definition(s).", app.c_str());
133  continue;
134  }
135  applications_defined.insert(app);
136 
137  list<string> args(tokens.begin() + 1, tokens.end());
138  applications.push_back(make_pair(app, args));
139  }
140 
141  return applications;
142 }

186 int daemon() {
187  switch (fork()) {
188  case -1:
189  /* Fork failed! */
190  return -1;
191 
192  case 0:
193  /* Daemon process */
194  break;
195 
196  default:
197  /* Valid PID returned, exit the parent process. */
198  exit(EXIT_SUCCESS);
199  break;
200  }
201 
202  /* Change the file mode mask */
203  umask(0);
204 
205  /* Create a new SID for the child process */
206  if (setsid() == -1) {
207  return -1;
208  }
209 
210  /* Close out the standard file descriptors */
211  int fd = open("/dev/null", O_RDWR, 0);
212  if (fd != -1) {
213  dup2(fd, STDIN_FILENO);
214  dup2(fd, STDOUT_FILENO);
215  dup2(fd, STDERR_FILENO);
216  if (fd > 2) {
217  close(fd);
218  }
219  }
220 
221  return 0;
222 }

560 static void finish_booting(Kernel* kernel, const Application_list& applications,
561  vector<string> interfaces, bool reliable) {
562  string fatal_error;
563 
564  try {
565  /* Boot the components defined on the command-line */
566  BOOST_FOREACH(Application_params app, applications) {
567  Application_name& name = app.first;
568  if (!kernel->get(name)) {
569  kernel->install(name, INSTALLED);
570  }
571  }
572  }
573  catch (const runtime_error& e) {
574  fatal_error = e.what();
575  }
576 
577  /* Report the installation results. */
578  lg.dbg("Application installation report:");
579  BOOST_FOREACH(Component_context* ctxt, kernel->get_all()) {
580  lg.dbg("%s:\n%s\n", ctxt->get_name().c_str(),
581  ctxt->get_status().c_str());
582  }
583 
584  /* Check all requested applications have booted. */
585  if (fatal_error != "") {
586  lg.err("%s", fatal_error.c_str());
587  exit(EXIT_FAILURE);
588  }
589 
590  try {
591  /* Connect the openflow interfaces */
592  BOOST_FOREACH (string& interface, interfaces) {
593  nox::connect(Openflow_connection_factory::create(interface),
594  reliable);
595  }
596  }
597  catch (const runtime_error& e) {
598  lg.err("%s", e.what());
599  exit(EXIT_FAILURE);
600  }
601 
602  nox::post_event(new Bootstrap_complete_event());
603  lg.info("nox bootstrap complete");
604 }
