56 /* In order to allow OFLib to process experimenter features, callback functions
57  * must be defined, and passed to the OFLib functions. Each experimenter
58  * feature has its own group of callbacks, and these groups are collected in a
59  * single ofl_exp structure.
60  * If an experimenter feature is not to be used, the callback can be NULLed.
61  *
62  * The internal representation of the experimenter structures can be freely
63  * defined, but they must use the relevant OFLib headers.
64  *
65  * For each feature the following functions must be provided:
66  *
67  * pack: given an experimenter structure, the function should return its
68  * OpenFlow wire format representation.
69  * unpack: given the OpenFlow wire format representation of an experimenter
70  * structure, it should return the experimenter structure. The passed
71  * in len tells the available length of bytes in the wire format. The
72  * function should subtract the amount of bytes used up during the
73  * conversion.
74  * free: passing an experimenter structure, this function must make sure the
75  * structure is freed.
76  * ofp_len: passing an experimenter structure, the function must return the
77  * length of the OpenFlow wire format representation of the structure.
78  * to_string: passing an experimenter structure, the function must return a
79  * string representation of the structure.
80  *
81  */
82

141 /* OFL_ERROR should be returned as ofl_err, when there was an error, but
142  * there is no appropriate OpenFlow error type defined, or when no error
143  * message should be generated because of the error. */
144 #define OFL_ERROR 0xffffffff

146 /* Creates an ofl_err from an OpenFlow error type and code */
147 static inline ofl_err
148 ofl_error(uint16_t type, uint16_t code) {
149  /* NOTE: highest bit is always set to one, so no error value is zero */
150  uint32_t ret = type;
151  return 0x80000000 | ret << 16 | code;
152 }
153 
154 /* Returns the error type of an ofl_err */
155 static inline uint16_t
156 ofl_error_type(ofl_err error) {
157  return (0x7fff0000 & error) >> 16;
158 }
159 
160 /* Returns the error code of an ofl_err */
161 static inline uint16_t
162 ofl_error_code(ofl_err error) {
163  return error & 0x0000ffff;
164 }
