52  Disposition handler(const Event& e)
53  {
54  const Ofp_msg_event& pi = assert_cast<const Ofp_msg_event&>(e);
55  struct ofl_msg_packet_in *in = (struct ofl_msg_packet_in *)**pi.msg;
56 
57 lg.dbg("***packetin in Hub");
58  Flow *flow = new Flow((struct ofl_match*) in->match);
59 
60  /* drop all LLDP packets */
61  uint16_t dl_type;
62  flow->get_Field("eth_type",&dl_type);
63  if (dl_type == ethernet::LLDP){
64  return CONTINUE;
65  }
66  Flow fl;
67  Actions *act = new Actions();
68  act->CreateOutput(OFPP_FLOOD);
69  act->CreateSetField("eth_type", &dl_type);
70  Instruction *inst = new Instruction();
71  inst->CreateApply(act);
72  FlowMod *mod = new FlowMod(0x00ULL,0x00ULL,0,OFPFC_ADD, 5, 5, OFP_DEFAULT_PRIORITY,
73  in->buffer_id, OFPP_ANY, OFPG_ANY,0x00000);
74  mod->AddInstructions(inst);
75  mod->AddMatch(&fl.match);
76  send_openflow_msg(pi.dpid, (struct ofl_msg_header *)&mod->fm_msg, 0/*xid*/, true/*block*/);
77 
78  if (in->buffer_id == UINT32_MAX) {
79  if (in->total_len == in->data_length) {
80  uint32_t in_port;
81  flow->get_Field("in_port", &in_port);
82  send_openflow_pkt(pi.dpid, Array_buffer(in->data, in->data_length), in_port, OFPP_FLOOD, true/*block*/);
83  } else {
84  /* Control path didn't buffer the packet and didn't send us
85  * the whole thing--what gives? */
86  lg.dbg("total_len=%zu data_len=%zu\n", in->total_len, in->data_length);
87  }
88  }
89 
90  return CONTINUE;
91  }
