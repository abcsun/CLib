35 /* Valid component states. */
36 enum Component_state {
37  NOT_INSTALLED = 0, /* Component has been discovered. */
38  DESCRIBED = 1, /* Component (XML) description has been
39  processed. */
40  LOADED = 2, /* Component library has been loaded, if
41  necessary. */
42  FACTORY_INSTANTIATED = 3, /* Component factory has been
43  instantiated from the library. */
44  INSTANTIATED = 4, /* Component itself has been
45  instantiated. */
46  CONFIGURED = 5, /* Component's configure() has been
47  called. */
48  INSTALLED = 6, /* Component's install() has been
49  called. */
50  ERROR = 7 /* Any state transition can lead to
51  error state, if an unrecovarable
52  error is occurs. */
53 };
