18 /* Netlink interface.
19  *
20  * Netlink is a datagram-based network protocol primarily for communication
21  * between user processes and the kernel, and mainly on Linux. Netlink is
22  * specified in RFC 3549, "Linux Netlink as an IP Services Protocol".
23  *
24  * Netlink is not suitable for use in physical networks of heterogeneous
25  * machines because host byte order is used throughout. */

145 /* A Netlink message.
146  *
147  * This is a wrapper for Buffer that is specialized for composing and parsing
148  * Netlink messages. Probably, much of its functionality should be generalized
149  * and integrated back into Buffer, but for now it seems to work OK.
150  *
151  * It is unclear whether a single class should be used for both composing and
152  * parsing messages, since these are fairly different usage patterns.
153  *
154  * The extensions beyond Buffer have these purposes:
155  *
156  * - Provide the ability to "pull" possibly missing headers. If they are
157  * missing, ok() will return false. (A bare Buffer has no "invalid"
158  * state, at the moment.)
159  *
160  * - Handle Netlink attributes.
161  *
162  * - Pad all pulls and puts with NLMSG_ALIGN (in practice, 4-byte
163  * alignment).
164  */

329 /* A Netlink message parsing policy.
330  *
331  * Specifies how to parse the collection of attributes that a Netlink message
332  * may contain as its payload.
333  *
334  * Example declaration (should be in namespace scope):
335  *
336  * std::tr1::array<Nl_attr, 3> dp_policy_elems = { {
337  * Nl_attr::u32(DP_GENL_A_DP_IDX).require(),
338  * Nl_attr::u32(DP_GENL_A_MC_GROUP).require(),
339  * Nl_attr::string(DP_GENL_A_PORTNAME, 1, IFNAMSIZ - 1),
340  * } };
341  *
342  * Nl_policy dp_policy(dp_policy_elems);
343  *
344  * Example usage:
345  *
346  * Genl_message msg(buffer);
347  * if (!msg.ok()) {
348  * ...handle error...
349  * }
350  *
351  * std::tr1::array<nlattr*, DP_GENL_A_MAX + 1> attrs;
352  * if (!dp_policy.parse(msg.rest(), attrs)) {
353  * ...handle error...
354  * }
355  *
356  */
357 class Nl_policy
358 {
359 public:
360  /* Constructs an Nl_policy with the N policy attributes in 'elements'.
361  *
362  * 'elements' must persist for the entire lifetime of the policy (see
363  * above for typical usage). */
364  template <size_t N>
365  Nl_policy(std::tr1::array<Nl_attr, N>& elements) {
366  init(&elements[0], N);
367  }
368 
369  /* Parses 'buffer' according to the policy, filling in each element of
370  * 'attrs' with a pointer to the corresponding attribute or (if the
371  * attribute is not found) a null pointer.
372  *
373  * Returns true if the policy is satisfied, false if it is violated. */
374  template <size_t N>
375  bool parse(const Buffer& buffer, std::tr1::array<nlattr*, N>& attrs) {
376  return parse(buffer, &attrs[0], N);
377  }
378 private:
379  std::vector<Nl_attr*> elements;
380  size_t n_required;
381 
382  void init(Nl_attr[], size_t n);
383  bool parse(const Buffer&, nlattr*[], size_t n);
384 };
