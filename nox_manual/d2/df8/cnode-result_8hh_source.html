23 /*
24  * Classifier result rule set holder.
25  *
26  * Templated over the Expr and Action types used by the Rules the result set
27  * will hold. Is also templated over the data type that the Rules' expressions
28  * should be checked for matches against (e.g. a packet, or another Expr, etc).
29  *
30  * Usage: Should be constructed with the data to classify, and then passed to
31  * Classifier::get_rules(), which will fill in the set's priority queue with
32  * matching rules. Matches at this point are only based on the tree traversal,
33  * and don't necessarily imply a full match, and thus iteration through the
34  * result set should be done with the next() method, which will skip over any
35  * rules in the priority queue that don't fully match 'data'.
36  */

102 void
103 Cnode_result<Expr, Action, Data>::push(const Rule_list& rules)
104 {
105  if (!rules.empty()) {
106  current_rule r(rules);
107  if (num_lists == traversed.size()) {
108  traversed.push_back(r);
109  } else {
110  traversed[num_lists] = r;
111  }
112  ++num_lists;
113  }

125 template<class Expr, typename Action, typename Data>
126 const Rule<Expr, Action>*
127 Cnode_result<Expr, Action, Data>::next()
128 {
129  const Rule<Expr, Action>* match = NULL;
130  uint32_t min_pri = 0;
131  uint32_t min_idx = 0;
132 
133  for (uint32_t i = 0; i < num_lists;) {
134  current_rule& current = traversed[i];
135  const Rule<Expr, Action>* rule = *(current.rule);
136  if (match == NULL || rule->priority < min_pri) {
137  if (current.ismatch || matches(rule->id, rule->expr, *data)) {
138  match = rule;
139  min_pri = match->priority;
140  min_idx = i;
141  current.ismatch = true;
142  ++i;
143  } else {
144  ++(current.rule);
145  if (current.rule == current.end) {
146  current = traversed[--num_lists];
147  }
148  }
149  } else {
150  ++i;
151  }
152  }
153 
154  if (match != NULL) {
155  current_rule& current = traversed[min_idx];
156  ++(current.rule);
157  if (current.rule == current.end) {
158  current = traversed[--num_lists];
159  } else {
160  current.ismatch = false;
161  }
162  }
163 
164  return match;
165 
166 }
