41 /* Initializes 'b' as an empty ofpbuf that contains the 'allocated' bytes of
42  * memory starting at 'base'.
43  *
44  * 'base' should ordinarily be the first byte of a region obtained from
45  * malloc(), but in circumstances where it can be guaranteed that 'b' will
46  * never need to be expanded or freed, it can be a pointer into arbitrary
47  * memory. */
48 void
49 ofpbuf_use(struct ofpbuf *b, void *base, size_t allocated)
50 {
51  b->base = b->data = base;
52  b->allocated = allocated;
53  b->size = 0;
54  b->l2 = b->l3 = b->l4 = b->l7 = NULL;
55  b->next = NULL;
56  b->private_p = NULL;
57 }

138 /* Returns the number of bytes of headroom in 'b', that is, the number of bytes
139  * of unused space in ofpbuf 'b' before the data that is in use. (Most
140  * commonly, the data in a ofpbuf is at its beginning, and thus the ofpbuf's
141  * headroom is 0.) */
142 size_t
143 ofpbuf_headroom(const struct ofpbuf *b)
144 {
145  return (char*)b->data - (char*)b->base;
146 }

156 /* Changes 'b->base' to 'new_base' and adjusts all of 'b''s internal pointers
157  * to reflect the change. */
158 static void
159 ofpbuf_rebase__(struct ofpbuf *b, void *new_base)
160 {
161  if (b->base != new_base) {
162  uintptr_t base_delta = (char*)new_base - (char*)b->base;
163  b->base = new_base;
164  b->data = (char*)b->data + base_delta;
165  if (b->l2) {
166  b->l2 = (char*)b->l2 + base_delta;
167  }
168  if (b->l3) {
169  b->l3 = (char*)b->l3 + base_delta;
170  }
171  if (b->l4) {
172  b->l4 = (char*)b->l4 + base_delta;
173  }
174  if (b->l7) {
175  b->l7 = (char*)b->l7 + base_delta;
176  }
177  }
178 }

180 /* Reallocates 'b' so that it has exactly 'new_tailroom' bytes of tailroom. */
181 static void
182 ofpbuf_resize_tailroom__(struct ofpbuf *b, size_t new_tailroom)
183 {
184  b->allocated = ofpbuf_headroom(b) + b->size + new_tailroom;
185  ofpbuf_rebase__(b, xrealloc(b->base, b->allocated));

188 /* Ensures that 'b' has room for at least 'size' bytes at its tail end,
189  * reallocating and copying its data if necessary. Its headroom, if any, is
190  * preserved. */
191 void
192 ofpbuf_prealloc_tailroom(struct ofpbuf *b, size_t size)
193 {
194  if (size > ofpbuf_tailroom(b)) {
