53 DSO_deployer::DSO_deployer(Kernel* kernel, const list<string>& lib_dirs_)
54  : Component(0), lib_dirs(lib_dirs_) {
55  using namespace boost::filesystem;
56 
57 #ifdef USE_LTDL
58  /* Initialize preloaded symbol table */
59  LTDL_SET_PRELOADED_SYMBOLS();
60 
61  /* Initialize the libtool dynamic library loading facilities */
62  if (::lt_dlinit()) {
63  throw runtime_error("lt_dlinit() failed: " +
64  demangle_undefined_symbol(::lt_dlerror()));
65  }
66 
67  if (!::lt_dlopen(0)) {
68  throw runtime_error("lt_dlopen() for the main() failed: " +
69  demangle_undefined_symbol(::lt_dlerror()));
70  }
71 #else
72  /* Initialize the dlopen() dynamic library loading facilities */
73  if (!::dlopen(0, RTLD_NOW | RTLD_GLOBAL)) {
74  throw runtime_error("dlopen() for the main() failed: " +
75  demangle_undefined_symbol(::dlerror()));
76  }
77 #endif
78 
79  list<path> description_files;
80  BOOST_FOREACH(string directory, lib_dirs) {
81  list<path> results = scan(directory);
82  description_files.insert(description_files.end(),
83  results.begin(), results.end());
84  }
85 
86  BOOST_FOREACH(path p, description_files) {
87  const string f = p.string();
88  path directory = p;
89  directory.remove_leaf();
90 
91  const json_object* d = json::load_document(f);
92  if (d->type == json_object::JSONT_NULL) {
93  lg.err("Can't load and parse '%s'", f.c_str());
94  continue;
95  }
96 
97  json_object* components = json::get_dict_value(d, "components");
98  json_array* componentList = (json_array*) components->object;
99 
100  json_array::iterator li;
101  for(li=componentList->begin(); li!=componentList->end(); ++li) {
102  try {
103  Component_context* ctxt =
104  new DSO_component_context(kernel, directory.string(), *li);
105  if (uninstalled_contexts.find(ctxt->get_name()) ==
106  uninstalled_contexts.end()) {
107  uninstalled_contexts[ctxt->get_name()] = ctxt;
108  } else {
109  lg.err("Component '%s' declared multiple times.",
110  ctxt->get_name().c_str());
111  delete ctxt;
112  }
113  } catch (const bad_cast& e) {
114  // Not a DSO component, skip.
115  continue;
116  }
117  }
118  }
119 
120  /* Cross-check they are no duplicate component definitions across
121  deployers. */
122  BOOST_FOREACH(const Deployer* deployer, kernel->get_deployers()) {
123  BOOST_FOREACH(Component_context* ctxt, deployer->get_contexts()) {
124  Component_name_context_map::iterator i =
125  uninstalled_contexts.find(ctxt->get_name());
126  if (i != uninstalled_contexts.end()) {
127  lg.err("Component '%s' declared multiple times.",
128  ctxt->get_name().c_str());
129  uninstalled_contexts.erase(i);
130  }
131  }
132  }
133 
134  /* Finally, register itself as a deployer responsible for DSO
135  components. */
136  kernel->attach_deployer(this);
137 }

164 DSO_component_context::DSO_component_context(Kernel* kernel,
165  const string& home_path,
166  json_object* description)
167  : Component_context(kernel) {
168  using namespace boost;
169  using namespace json;
170 
171  install_actions[DESCRIBED] = bind(&DSO_component_context::describe, this);
172  install_actions[LOADED] = bind(&DSO_component_context::load, this);
173  install_actions[FACTORY_INSTANTIATED] =
174  bind(&DSO_component_context::instantiate_factory, this);
175  install_actions[INSTANTIATED] =
176  bind(&DSO_component_context::instantiate, this);
177  install_actions[CONFIGURED] = bind(&DSO_component_context::configure, this);
178  install_actions[INSTALLED] =
179  bind(&DSO_component_context::install, this);
180 
181  /* Determine the configuration */
182  json_object* attr;
183 
184  attr = json::get_dict_value(description, "name");
185  name = attr->get_string(true);
186 
187  attr = json::get_dict_value(description, "library");
188  if (attr==NULL) {
189  throw bad_cast();
190  }
191  library = attr->get_string(true);
192 
193  if (library.length() > 3 && library.find(".so") == library.length() - 3) {
194  lg.warn("Dropped an unneccessary '.so' suffix in a shared library "
195  "file definition: %s", library.c_str());
196  library = library.substr(0, library.size() - 3);
197  }
198 
199  this->home_path = home_path;
200 
201  attr = json::get_dict_value(description, "dependencies");
202  if (attr!=NULL) {
203  json_array* depList = (json_array*) attr->object;
204  for(json_array::iterator li=depList->begin(); li!=depList->end(); ++li){
205  dependencies.push_back(new Name_dependency(((json_object*)*li)->get_string(true)));
206  }
207  }
208 
209  configuration = new Component_configuration(description,
210  kernel->get_arguments(name));
211  json_description = description;
212 }

220 void
221 DSO_component_context::load() {
222  const char* error_msg;
223  handle = open_library((home_path + "/" + library).c_str(), &error_msg);
224  string error(demangle_undefined_symbol(error_msg));
225 
226  /* A little extra check for libtool build directory */
227  if (!handle) {
228  handle =
229  open_library((home_path + "/.libs/" + library).c_str(), &error_msg);
230  error = "'" + error + "' or '"+demangle_undefined_symbol(error_msg)+"'";
231  }
232 
233  if (!handle) {
234  current_state = ERROR;
235  error_message = "Can't open a dynamic library: " + error;
236  } else {
237  current_state = LOADED;
238  }
239 }

254  string v = s;

265 void
266 DSO_component_context::instantiate_factory() {
267  component_factory_function* f = 0;
268 
269  /* Prefer a factory function with the embedded component name, but if
270  that's not found, default to one without the name. */
271  string function_with_component_name =
272  replace(library, '-', "_") + "_get_factory";
273 
274  f = find_factory_function(function_with_component_name.c_str());
275  if (!f) { f = find_factory_function("get_factory"); }
276  if (!f) {
277  current_state = ERROR;
278  error_message = library + " does not implement " +
279  function_with_component_name + "() nor get_factory() function";
280  return;
281  }
282 
283  factory = f();
284  interface = factory->get_interface();
285  current_state = FACTORY_INSTANTIATED;
286 }
