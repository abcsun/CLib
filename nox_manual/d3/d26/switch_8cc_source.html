207 int extract_int_from_buf(int *ret, uint8_t * data, int start, int size, uint8_t separator )
208 { //'start' starts from 0, and size is the buf's size
209  int tmp=0;
210  if(start>=size)
211  return -1; //-1 when start>end
212  while(data[start] != separator && start<size)
213  {
214  tmp *= 10;
215  tmp += (data[start]-48);
216  start++;
217  }
218  *ret = tmp;
219  if(data[start] == separator)
220  return start+1;
221  else
222  return start;
223 }

319 {
320  VLOG_DBG(log,"TCP socket connection accepted by demo_tcp_connection");
321  Array_buffer buf(1024);
322  bool running = true;
323  int dataSize = 0;
324  uint8_t *data;
325  while(running)
326  {
327  sock->read_wait();
328  co_block();
329  dataSize = sock->read(buf, false);
330  data = buf.data();
331  if(dataSize<=0)
332  {
333  running = false;
334  }
335  processBlock(buf, dataSize);
336 
337  //VLOG_DBG(log,"recv:%d B", dataSize);
338  }
339  int error = sock->close();
340  if(error)
341  VLOG_DBG(log, "demo_tcp_connection close with error: %d", error);
342  VLOG_DBG(log, "socket closed");
