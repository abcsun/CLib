95 /* Critical sections.
96  *
97  * A critical section is customarily a region of code that can only be executed
98  * by one thread of execution at a time. With cooperative threads, every
99  * section of code between calls to blocking functions is a critical section.
100  * Thus, there's a reduced need for the synchronization primitives used to
101  * define critical sections.
102  *
103  * It's easy, however, to accidentally insert a call to a blocking function
104  * into code that must execute as a critical section. Thus, these functions,
105  * which allow for assertions that a region of code will not block. Call
106  * co_enter_critical_section() to increase the "critical section count",
107  * co_exit_critical_section() to decrease it, and co_might_yield() to assert
108  * that the count is 0.
109  */
110 #ifndef NDEBUG
111 void co_enter_critical_section(void);
112 void co_exit_critical_section(void);
113 void co_might_yield(void);
114 void co_might_yield_if(bool block);
115 #else
116 #define co_enter_critical_section() do { } while (0)
117 #define co_exit_critical_section() do { } while (0)
118 #define co_might_yield() do { } while (0)
119 #define co_might_yield_if(BLOCK) do { } while (0)
120 #endif

133 /* Waiting for threads to die.
134  *
135  * There is no 'join' function, but:
136  *
137  * - A native thread can wait for another native thread to complete by
138  * downing a Native_sema passing into the joined thread's initial function.
139  *
140  * - A native thread can wait for a cooperative thread or FSM to complete by
141  * migrating into the thread or FSM's group, then using
142  * co_join_completion(), and waiting on the completion.
143  *
144  * - A cooperative thread or FSM can wait for another cooperative thread or
145  * FSM by using co_join_completion() and waiting on the completion.
146  *
147  * - A cooperative thread can wait for a native thread to complete by
148  * migrating itself out of its thread group (becoming a native thread) and
149  * downing a Native_sema passing into the joined thread's initial function.
150  *
151  * - An FSM cannot (easily) wait for a native thread to complete.
152  *
153  * Don't use pthread_join() to wait for native threads: there is a race
154  * between obtaining the thread's pthread_t and the destruction of the
155  * co_thread following the thread's death.
156  */
157 void co_join_completion(co_thread *, Co_completion*);

165 /* Blocking.
166  *
167  * Before blocking, call one or more "wait functions" that register an event to
168  * wait one. By convention, wait function names end in "_wait". co_block()
169  * will return when at least one of the registered events has occurred. (If
170  * co_block() is called without any events being registered, the thread will
171  * block forever.)
172  *
173  * Calling a wait function enters a critical section. co_block() in turn exits
174  * one critical section for each per registered event. Thus, between calling a
175  * wait function and calling co_block(), no blocking functions may be called.
176  */
177 void co_block(void);

199 /* Waitqueue.
200  *
201  * A waitqueue is a low-level data structure used as a building block for
202  * synchronization primitives and other forms of events. If you are not
203  * building your own synchronization primitives, you do not need to understand
204  * them.
205  *
206  * A waitqueue is simply a singly linked list of events used as a FIFO queue.
207  * Each event contains a linked list node, so no dynamic allocation is needed.
208  *
209  * The implementation of waitqueues is too clever: it avoids initializing the
210  * 'next' field of the new event to NULL in co_wq_add_event() or the 'head'
211  * field of co_waitqueue to NULL in co_wq_init() by making the *next* call to
212  * co_wq_wake_all() do it. Anything that traverses a waitqueue looking for a
213  * null pointer must thus terminate the list with a null pointer by setting
214  * *wq->tail to NULL. */
215 class Co_waitqueue
216  : boost::noncopyable
217 {
218 public:
219  Co_waitqueue();
220  bool empty() const;
221  void wake_all(int retval);
222  size_t wake_n(int reval, size_t max_wake);
223  co_thread* wake_1(int retval);
224  void wait(int* retvalp);
225  void dequeue(co_event*);
226  void requeue(co_event*);
227 private:
228  co_event *head; /* First event; not necessarily initialized. */
229  co_event **tail; /* Address of 'next' field of last event, or in
230  * empty waitqueue address of 'head'. */
231 };
