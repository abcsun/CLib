49 ofl_structs_instructions_unpack(struct ofp_instruction *src, size_t *len, struct ofl_instruction_header **dst, struct ofl_exp *exp) {
50  size_t ilen;
51  struct ofl_instruction_header *inst = NULL;
52 
53  if (*len < sizeof(struct ofp_instruction)) {
54  OFL_LOG_WARN(LOG_MODULE, "Received instruction is too short (%zu).", *len);
55  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
56  }
57 
58  if (*len < ntohs(src->len)) {
59  OFL_LOG_WARN(LOG_MODULE, "Received instruction has invalid length (set to %u, but only %zu received).", ntohs(src->len), *len);
60  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
61  }
62  ilen = ntohs(src->len);
63 
64  switch (ntohs(src->type)) {
65  case OFPIT_GOTO_TABLE: {
66  struct ofp_instruction_goto_table *si;
67  struct ofl_instruction_goto_table *di;
68 
69  if (ilen < sizeof(struct ofp_instruction_goto_table)) {
70  OFL_LOG_WARN(LOG_MODULE, "Received GOTO_TABLE instruction has invalid length (%zu).", *len);
71  return ofl_error(OFPET_BAD_ACTION, OFPBRC_BAD_LEN);
72  }
73 
74  si = (struct ofp_instruction_goto_table *)src;
75 
76  if (si->table_id == 0xff) {
77  if (OFL_LOG_IS_WARN_ENABLED(LOG_MODULE)) {
78  char *ts = ofl_table_to_string(si->table_id);
79  OFL_LOG_WARN(LOG_MODULE, "Received GOTO_TABLE instruction has invalid table_id (%s).", ts);
80  free(ts);
81  }
82  return ofl_error(OFPET_BAD_INSTRUCTION, OFPBIC_BAD_TABLE_ID);
83  }
84 
85  di = (struct ofl_instruction_goto_table *)malloc(sizeof(struct ofl_instruction_goto_table));
86 
87  di->table_id = si->table_id;
88 
89  inst = (struct ofl_instruction_header *)di;
90  ilen -= sizeof(struct ofp_instruction_goto_table);
91  break;
92  }
93 
94  case OFPIT_WRITE_METADATA: {
95  struct ofp_instruction_write_metadata *si;
96  struct ofl_instruction_write_metadata *di;
97 
98  if (ilen < sizeof(struct ofp_instruction_write_metadata)) {
99  OFL_LOG_WARN(LOG_MODULE, "Received WRITE_METADATA instruction has invalid length (%zu).", *len);
100  return ofl_error(OFPET_BAD_ACTION, OFPBRC_BAD_LEN);
101  }
102 
103  si = (struct ofp_instruction_write_metadata *)src;
104  di = (struct ofl_instruction_write_metadata *)malloc(sizeof(struct ofl_instruction_write_metadata));
105 
106  di->metadata = ntoh64(si->metadata);
107  di->metadata_mask = ntoh64(si->metadata_mask);
108 
109  inst = (struct ofl_instruction_header *)di;
110  ilen -= sizeof(struct ofp_instruction_write_metadata);
111  break;
112  }
113  case OFPIT_WRITE_ACTIONS:
114  case OFPIT_APPLY_ACTIONS: {
115  struct ofp_instruction_actions *si;
116  struct ofl_instruction_actions *di;
117  struct ofp_action_header *act;
118  ofl_err error;
119  size_t i;
120 
121  if (ilen < sizeof(struct ofp_instruction_actions)) {
122  OFL_LOG_WARN(LOG_MODULE, "Received *_ACTIONS instruction has invalid length (%zu).", *len);
123  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
124  }
125  ilen -= sizeof(struct ofp_instruction_actions);
126 
127  si = (struct ofp_instruction_actions *)src;
128  di = (struct ofl_instruction_actions *)malloc(sizeof(struct ofl_instruction_actions));
129 
130  error = ofl_utils_count_ofp_actions((uint8_t *)si->actions, ilen, &di->actions_num);
131  if (error) {
132  free(di);
133  return error;
134  }
135  di->actions = (struct ofl_action_header **)malloc(di->actions_num * sizeof(struct ofl_action_header *));
136 
137  act = si->actions;
138  for (i = 0; i < di->actions_num; i++) {
139  error = ofl_actions_unpack(act, &ilen, &(di->actions[i]), exp);
140  if (error) {
141  *len = *len - ntohs(src->len) + ilen;
142  OFL_UTILS_FREE_ARR_FUN2(di->actions, i,
143  ofl_actions_free, exp);
144  free(di);
145  return error;
146  }
147  act = (struct ofp_action_header *)((uint8_t *)act + ntohs(act->len));
148  }
149 
150  inst = (struct ofl_instruction_header *)di;
151  break;
152  }
153  case OFPIT_CLEAR_ACTIONS: {
154  if (ilen < sizeof(struct ofp_instruction_actions)) {
155  OFL_LOG_WARN(LOG_MODULE, "Received CLEAR_ACTIONS instruction has invalid length (%zu).", *len);
156  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
157  }
158 
159  inst = (struct ofl_instruction_header *)malloc(sizeof(struct ofl_instruction_header));
160  inst->type = (enum ofp_instruction_type)ntohs(src->type);
161 
162  ilen -= sizeof(struct ofp_instruction_actions);
163  break;
164  }
165  case OFPIT_METER: {
166  struct ofp_instruction_meter *si;
167  struct ofl_instruction_meter *di;
168 
169  if (ilen < sizeof(struct ofp_instruction_meter)) {
170  OFL_LOG_WARN(LOG_MODULE, "Received METER instruction has invalid length (%zu).", *len);
171  return ofl_error(OFPET_BAD_ACTION, OFPBRC_BAD_LEN);
172  }
173  si = (struct ofp_instruction_meter*)src;
174  di = (struct ofl_instruction_meter *)malloc(sizeof(struct ofl_instruction_meter));
175 
176  di->meter_id = ntohl(si->meter_id);
177 
178  inst = (struct ofl_instruction_header *)di;
179  ilen -= sizeof(struct ofp_instruction_meter);
180  break;
181  }
182  case OFPIT_EXPERIMENTER: {
183  ofl_err error;
184 
185  if (exp == NULL || exp->inst == NULL || exp->inst->unpack == NULL) {
186  OFL_LOG_WARN(LOG_MODULE, "Received EXPERIMENTER instruction, but no callback was given.");
187  return ofl_error(OFPET_BAD_INSTRUCTION, OFPBIC_UNSUP_INST);
188  }
189  error = exp->inst->unpack(src, &ilen, &inst);
190  if (error) {
191  return error;
192  }
193  break;
194  }
195  }
196 
197  // must set type before check, so free works correctly
198  inst->type = (enum ofp_instruction_type)ntohs(src->type);
199 
200  if (ilen != 0) {
201  *len = *len - ntohs(src->len) + ilen;
202  OFL_LOG_WARN(LOG_MODULE, "The received instruction contained extra bytes (%zu).", ilen);
203  ofl_structs_free_instruction(inst, exp);
204  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
205  }
206 
207  *len -= ntohs(src->len);
208  (*dst) = inst;
209 
210  return 0;
211 }

412 ofl_structs_bucket_unpack(struct ofp_bucket *src, size_t *len, uint8_t gtype, struct ofl_bucket **dst, struct ofl_exp *exp) {
413  struct ofl_bucket *b;
414  struct ofp_action_header *act;
415  size_t blen;
416  ofl_err error;
417  size_t i;
418 
419  if (*len < sizeof(struct ofp_bucket)) {
420  OFL_LOG_WARN(LOG_MODULE, "Received bucket is too short (%zu).", *len);
421  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
422  }
423 
424  if (*len < ntohs(src->len)) {
425  OFL_LOG_WARN(LOG_MODULE, "Received bucket has invalid length (set to %u, but only %zu received).", ntohs(src->len), *len);
426  return ofl_error(OFPET_BAD_ACTION, OFPBAC_BAD_LEN);
427  }
428 
429  blen = ntohs(src->len) - sizeof(struct ofp_bucket);
430 
431  if (gtype == OFPGT_SELECT && ntohs(src->weight) == 0) {
432  OFL_LOG_WARN(LOG_MODULE, "Received bucket has no weight for SELECT group.");
433  return ofl_error(OFPET_GROUP_MOD_FAILED, OFPGMFC_INVALID_GROUP);
434  }
435 
436  if (gtype != OFPGT_SELECT && ntohs(src->weight) > 0) {
437  OFL_LOG_WARN(LOG_MODULE, "Received bucket has weight for non-SELECT group.");
438  return ofl_error(OFPET_GROUP_MOD_FAILED, OFPGMFC_INVALID_GROUP);
439  }
440 
441  b = (struct ofl_bucket *)malloc(sizeof(struct ofl_bucket));
442 
443  b->weight = ntohs(src->weight);
444  b->watch_port = ntohl(src->watch_port);
445  b->watch_group = ntohl(src->watch_group);
446 
447  error = ofl_utils_count_ofp_actions((uint8_t *)src->actions, blen, &b->actions_num);
448  if (error) {
449  free(b);
450  return error;
451  }
452  b->actions = (struct ofl_action_header **)malloc(b->actions_num * sizeof(struct ofl_action_header *));
453 
454  act = src->actions;
455  for (i = 0; i < b->actions_num; i++) {
456  error = ofl_actions_unpack(act, &blen, &(b->actions[i]), exp);
457  if (error) {
458  *len = *len - ntohs(src->len) + blen;
459  OFL_UTILS_FREE_ARR_FUN2(b->actions, i,
460  ofl_actions_free, exp);
461  free(b);
462  return error;
463  }
464  act = (struct ofp_action_header *)((uint8_t *)act + ntohs(act->len));
465  }
466 
467  if (blen >= 8) {
468  *len = *len - ntohs(src->len) + blen;
469  ofl_structs_free_bucket(b, exp);
470  OFL_LOG_WARN(LOG_MODULE, "Received bucket has more than 64 bit padding (%zu).", blen);
471  return ofl_error(OFPET_BAD_REQUEST, OFPBRC_BAD_LEN);
472  }
473  *len -= ntohs(src->len);
474 
475  *dst = b;
476  return 0;
477 }
